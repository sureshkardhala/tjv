CREATE TABLE `posts` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `title` varchar(500) NOT NULL,
  `image` varchar(100) NOT NULL,
  `description` text NOT NULL,
  `slug_url` varchar(500) NOT NULL,
  `category` varchar(250) NOT NULL,
  `topic` varchar(100) NOT NULL,
  `seo_description` varchar(1000) NOT NULL,
  `seo_keywords` varchar(1000) NOT NULL,
  `user` varchar(500) NOT NULL,
  `updated_on` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(),
  `uploaded_on` timestamp NOT NULL DEFAULT current_timestamp()
) ENGINE=InnoDB DEFAULT CHARSET=latin1;



CREATE TABLE `users` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `username` varchar(255) NOT NULL,
  `email` varchar(255) NOT NULL,
  `password` varchar(255) NOT NULL,
  `status` varchar(100) NOT NULL,
  `otp` int(11) NOT NULL,
  `type` varchar(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=latin1;


CREATE TABLE `contact` (
  `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `name` varchar(500) NOT NULL,
  `email` varchar(100) NOT NULL,
  `phone` varchar(100) NOT NULL,
  `category` varchar(100) NOT NULL,
  `message` text NOT NULL,
  `uploaded_on` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp()
) ENGINE=InnoDB DEFAULT CHARSET=latin1;



CREATE TABLE `events` ( `id` int(11) NOT NULL AUTO_INCREMENT PRIMARY KEY, `title` varchar(500) NOT NULL, `image` varchar(100) NOT NULL, `description` text NOT NULL, `speaker` varchar(1000) NOT NULL, `speakerimage` varchar(100) NOT NULL, `event_date` varchar(250) NOT NULL, `event_time` varchar(250) NOT NULL, `speakerabout` text NOT NULL, `slug_url` varchar(500) NOT NULL, `category` varchar(250) NOT NULL, `topic` varchar(100) NOT NULL, `seo_description` varchar(1000) NOT NULL, `seo_keywords` varchar(1000) NOT NULL, `user` varchar(500) NOT NULL, `youtube` varchar(1000) NOT NULL, `facebook` varchar(1000) NOT NULL, `zoom` varchar(500) NOT NULL, `updated_on` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(), `uploaded_on` timestamp NOT NULL DEFAULT current_timestamp() )ENGINE=InnoDB DEFAULT CHARSET=latin1;
